                post
                /path-ai/tanya-tutor
            

            

                Endpoint untuk bertanya ke AI tutor. Mendukung percakapan
                multi-turn dengan konteks materi.
            

            
Argumen

            

                

                    nama
                    deskripsi
                    tipe
                

                

                    student_message
                    Pesan/pertanyaan dari siswa (min. 1 karakter, wajib)
                    string
                

                

                    conversation_history
                    Riwayat percakapan sebelumnya (opsional)
                    list[dict]
                

                

                    context_material
                    Materi konteks untuk tutor (opsional)
                    string
                

                

                    subject
                    Mata pelajaran (opsional)
                    string
                

                

                    topic
                    Topik pembahasan (opsional)
                    string
                

            

            
Contoh Response

            
{
    "reply": "Tentu! Fotosintesis adalah proses ...",
    "metadata": { ... }
}

        


        

            

            

                post
                /path-ai/buat-soal-quiz
            

            

                Generate soal quiz berdasarkan materi yang diberikan.
                Mendukung berbagai tipe soal dan tingkat kesulitan.
            

            
Argumen

            

                

                    nama
                    deskripsi
                    tipe
                

                

                    content
                    Materi untuk membuat soal (min. 10 karakter, wajib)
                    string
                

                

                    num_questions
                    Jumlah soal, range 1-20 (default: 5)
                    integer
                

                

                    question_type
                    Tipe soal: multiple_choice, true_false, essay,
                        short_answer
                    
                    string
                

                

                    difficulty
                    Tingkat kesulitan: easy, medium, hard
                    string
                

                

                    topic
                    Topik materi (opsional)
                    string
                

                

                    skip_validation
                    Skip validasi LLM (default: false)
                    boolean
                

            

            
Contoh Response

            
{
    "quiz": {
        "title": "Quiz Fotosintesis",
        "subject": "",
        "topic": "Fotosintesis",
        "total_questions": 5,
        "questions": [
            {
                "question_id": 1,
                "question": "Apa hasil utama fotosintesis?",
                "question_type": "multiple_choice",
                "options": [
                    { "label": "A", "text": "Glukosa", "is_correct": true },
                    { "label": "B", "text": "Protein", "is_correct": false }
                ],
                "correct_answer": "A",
                "explanation": "...",
                "difficulty": "medium",
                "concept_tag": "fotosintesis"
            }
        ],
        "summary": "..."
    },
    "metadata": { ... }
}

        


        

            

            

                post

                Content-Type: multipart/form-data.
            

            
Argumen

            

                

                    nama
                    deskripsi
                    tipe
                

                

                    file
                    File dokumen yang akan diekstrak (wajib)
                    UploadFile (multipart)
                

            

            
Contoh Response

            
{
    "text": "Isi teks yang diekstrak dari dokumen...",
    "text_length": 4949,
    "image_count": 3,
    "page_count": 10,
    "source": "materi.pdf"
}

                post
                /path-ai/nilai-semua-jawaban

                    deskripsi
                    tipe

                    list[dict]

        "needs_remedial": true,
        "score": 60.0,
        "threshold": 75.0,
        "weak_concepts": ["fotosintesis"],
        "misconceptions": [ ... ],
        "recommended_action": "ai_summary_card"
    },
    "metadata": { ... }
}

        


        

            

            

                post
                /path-ai/sederhanakan-materi
            

            

                Menyederhanakan materi sesuai jenjang target.
                Mode simplifier untuk jalur kuning (butuh dukungan),
                mode fast-track untuk jalur hijau.
            

            
Argumen

            

                

                    nama
                    deskripsi
                    tipe
                

                

                    content
                    Materi yang akan disederhanakan (min. 10 karakter, wajib)
                    string
                

                

                    mode
                    simplifier (jalur kuning) atau fast-track (jalur hijau)
                    string
                

                

                    target_level
                    Jenjang target: SD, SMP, SMA (default: SMA)
                    string
                

                

                    topic
                    Topik materi (opsional)
                    string
                

                

                    weak_concepts
                    Konsep lemah dari diagnostic, hanya untuk mode simplifier (opsional)
                    list[string]
                

            

            
Contoh Response

            
{
    "simplified_content": "Materi yang sudah disederhanakan...",
    "metadata": { ... }
}

                post
                /path-ai/buat-remedial
            

            

                Generate konten remedial untuk siswa yang skornya di bawah threshold.
                Jika skor di atas threshold, endpoint akan skip dan tidak generate konten.
            

            
Argumen

            

                

                    nama
                    deskripsi
                    tipe
                

                

                    score
                    Skor siswa, range 0-100 (wajib)
                    float
                

                

                    threshold
                    Batas minimum lulus (default: 75.0)
                    float
                

                

                    weak_concepts
                    List konsep yang lemah (opsional)
                    list[string]
                

                

                    misconceptions
                    List miskonsepsi: {concept, student_thinking, correct_understanding, severity}
                    list[dict]
                

                

                    original_content
                    Materi asli sebagai referensi (opsional)
                    string
                

                

                    student_level
                    Jenjang siswa (default: SMA)
                    string
                

            

            
Contoh Response

            
Jika skor < threshold:

            
{
    "remedial_content": "Konten remedial yang di-generate...",
    "metadata": { ... }
}

            
Jika skor >= threshold:

            
{
    "remedial_content": "",
    "skipped": true
}

Error Responses

            
Semua endpoint dapat mengembalikan error berikut:

            

                

                    kode status
                    isi
                    alasan
                

                

                    400
                    Bad Request
                    Input tidak valid atau file tidak didukung
                

                

                    422
                    Validation Error
                    Field wajib tidak diisi atau format salah
                

                

                    429
                    Model sedang sibuk (rate limited)
                    Terlalu banyak request ke AI model
                

                

                    500
                    Internal Server Error
                    Gagal generate / proses di server
                

                

                    502
                    Gagal menghubungi AI model
                    Error dari OpenRouter API
                

                

                    503
                    Model sedang tidak tersedia
                    AI model down / maintenance
